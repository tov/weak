8 /// Controls how a weak hash table uses its weak elements.
9 ///
10 /// In particular, specializations of this struct provide a uniform interface
11 /// for dealing with both `std::weak_ptr` and the various weak pairs provided
12 /// by this library.
13 template <class T>
14 struct weak_traits
15 {
16  /// Owns direct data, `shared_ptr` to indirect data.
17  ///
18  /// This type guarantees the presence of a key. It should be used when
19  /// inserting into a weak hash table.
20  using strong_type = typename T::strong_type;
21 
22  /// Borrows direct data, `shared_ptr` to indirect data.
23  ///
24  /// This type does not guarantee the presence of a key. It is used when
25  /// viewing the contents of a weak hash table.
26  using view_type = typename T::view_type;
27 
28  /// Const-borrows direct data, `shared_ptr` to indirect data.
29  ///
30  /// Like `view_type`, but for when viewing from a `const_iterator`.
31  using const_view_type = typename T::const_view_type;
32 
33  /// The key type.
34  ///
35  /// For pairs, this is the `first` component. For plain `weak_ptr<T>`,
36  /// this is `T`.
37  using key_type = typename T::key_type;
38 
39  /// Projects a key pointer from a `view_type` or `strong_type`.
40  ///
41  /// If given a `view_type`, the result might be `nullptr`.
42  template <class U>
43  static const key_type* key(const U& view)
44  {
45  return T::key(view);
46  }
47 
48  /// Steals a `view_type`, turning it into a `strong_type`.
49  ///
50  /// *UNCHECKED PRECONDITION*: the `view_type` is not expired.
51  static strong_type move(view_type& view)
52  {
53  return T::move(view);
54  }
55 };
