11 /// A map whose range values are stored by `std::weak_ptr`s.

30  /// Proxy class returned by `operator[](const Key&)`.
31  ///
32  /// Assigning to this proxy will assign to the value pointer in the map.
33  class proxy
34  {
35  public:
36  /// Copy constructor.
37  proxy(const proxy&) noexcept = default;
38  /// Move constructor.
39  proxy(proxy&&) noexcept = default;

43  /// No move-assignment.
44  proxy& operator=(proxy&&) = delete;

46  /// Dereferencing the proxy acts like dereferencing the `shared_ptr`.
47  T& operator*() const noexcept
48  {

50  }

53  std::shared_ptr<T> operator->() const noexcept

59  operator std::shared_ptr<T>() const noexcept

64  /// Assigning a `shared_ptr` to the proxy assigns into the map.
65  proxy& operator=(const std::shared_ptr<T>& value)
66  {
67  bucket_.value().second = value_ptr_ = value;

90  /// Looks up the given key in the hash table, returning a proxy for
91  /// the `shared_ptr` to the value.
92  ///
93  /// If the key doesn't exist then it is inserted and temporarily mapped
94  /// to an expired pointer. However, assigning a `shared_ptr` to the proxy
95  /// will stored the `shared_ptr` in the map instead.
96  proxy operator[](const Key& key)
97  {
98  Bucket* result_bucket;
99 
100  BaseClass::insert_helper_(
101  key,
102  [&](Bucket& bucket) {
103  BaseClass::construct_bucket_(bucket, key,
104  std::shared_ptr<T>{});
105  result_bucket = &bucket;
106  },
107  [&](Bucket& bucket) {
108  bucket.value().first = key;
109  bucket.value().second = std::shared_ptr<T>{};
110  result_bucket = &bucket;
111  },
112  [&](Bucket& bucket) {
113  result_bucket = &bucket;
114  });
115 
116  return proxy(*result_bucket);
117  }
118 };

128 /// Is `a` a submap of `b`?
129 ///
130 /// That is, are all the keys of `a` keys of `b`, and all the values equal
131 /// according to `compare`. Function `compare` defaults to equality, but
132 /// other relations are possible.
133 template <class Key, class T, class Hash, class KeyEqual, class Allocator,
134  class TEqual = std::equal_to<>>
135 bool submap(
136  const weak_value_unordered_map<Key, T, Hash, KeyEqual, Allocator>& a,
137  const weak_value_unordered_map<Key, T, Hash, KeyEqual, Allocator>& b,
138  TEqual compare = TEqual())
139 {
140  for (const auto& elem : a) {
141  auto iter = b.find(elem.first);
142  if (iter == b.end() || !compare(*(*iter).second, *elem.second))
143  return false;
144  }
145 
146  return true;
147 }
