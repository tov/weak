15 /// \mainpage weak: weak hash tables for C++17
16 ///
17 /// ## Get It
18 ///
19 /// The latest version is available [on GitHub](https://github.com/tov/weak).
20 ///
21 /// ## Use It
22 /// This library provides four forms of weak hash tables:
23 ///
24 /// - `weak_unordered_set`, which stores a set of `std::weak_ptr`s.
25 ///
26 /// - `weak_key_unordered_map`, which maps `std::weak_ptr`s to values.
27 ///
28 /// - `weak_value_unordered_map`, which maps keys to `std::weak_ptr`s.
29 ///
30 /// - `weak_weak_unordered_map`, which maps `std::weak_ptr`s to
31 /// `std::weak_ptr`s.
32 ///
33 /// Most of the interfaces of all four classes are common, and documented as
34 /// part of a shared base class `weak_hash_table_base`. All the constructors
35 /// may be found in that class as well.
36 ///
37 /// Here is a tiny example:
38 ///
39 /// ```cpp
40 /// weak_key_unordered_map<string, int> map;
41 /// CHECK( !map.member("hello") );
42 ///
43 /// auto hello = make_shared<string>("hello");
44 /// map.insert({hello, 5});
45 /// CHECK( map.member("hello") );
46 ///
47 /// hello = nullptr;
48 /// CHECK( !map.member("hello") );
49 /// ```
50 ///
51 /// For a more realistic example, see
52 /// [`intern_table.h`](https://github.com/tov/weak/blob/master/test/intern_table.h)
53 /// and
54 /// [`intern_table.cpp`](https://github.com/tov/weak/blob/master/test/intern_table.cpp).
55 ///

57 /// A weak Robin Hood hash table.
58 ///
59 /// Provides the common functionality for all the weak hash tables, including
60 /// both sets and maps.
61 ///
62 /// This is designed to work with element types that implement the
63 /// `weak_traits` interface. This includes `weak_ptr<T>` and the various weak
64 /// pair types provided by this library. Using a weak pair results in a weak
65 /// map, whereas using `weak_ptr<T>` results in a weak set. Usually this
66 /// should be instantiated through one of the derived classes, and this class
67 /// should not be used directly.
68 template <
69  class T,
70  class Hash = std::hash<typename weak_traits<T>::key_type>,
71  class KeyEqual = std::equal_to<>,
72  class Allocator = std::allocator<T>
73 >

581  template <class KeyLike>
582  const Bucket* lookup_(const KeyLike& key) const
583  {
584  size_t hash_code = hash_(key);
585  size_t pos = which_bucket_(hash_code);
586  size_t dist = 0;
587 
588  for (;;) {
589  const Bucket& bucket = buckets_[pos];
590  if (!bucket.used_)
591  return nullptr;
592 
593  if (dist > probe_distance_(pos, which_bucket_(bucket.hash_code_)))
594  return nullptr;
595 
596  if (hash_code == bucket.hash_code_) {
597  auto bucket_value_locked = bucket.value_.lock();
598  if (const auto* bucket_key = weak_trait::key(bucket_value_locked))
599  if (equal_(key, *bucket_key))
600  return &bucket;
601  }
602 
603  pos = next_bucket_(pos);
604  ++dist;
605  }
606  }

616  /// Places `value` in the table, starting at `pos` and moving forward.
617  void steal_(size_t hash_code, size_t pos, strong_value_type&& value)
618  {
619  size_t dist = probe_distance_(pos, which_bucket_(hash_code));
620 
621  for (;;) {
622  Bucket& bucket = buckets_[pos];
623 
624  if (!bucket.used_) {
625  construct_bucket_(bucket, std::move(value));
626  bucket.hash_code_ = hash_code;
627  return;
628  }
629 
630  if (bucket.value_.expired()) {
631  bucket.value_ = std::move(value);
632  bucket.hash_code_ = hash_code;
633  return;
634  }
635 
636  size_t existing_distance =
637  probe_distance_(pos, which_bucket_(bucket.hash_code_));
638  if (dist > existing_distance) {
639  auto bucket_locked = bucket.value_.lock();
640  bucket.value_ = std::exchange(value, weak_trait::move(bucket_locked));
641  // swap doesn't work because bitfield:
642  bucket.hash_code_ = std::exchange(hash_code, size_t(bucket.hash_code_));
643  dist = existing_distance;
644  }
645 
646  pos = next_bucket_(pos);
647  ++dist;
648  }
649  }

702 
703  /// Expects to call exactly one of the parameters `on_uninit`, `on_init`,
704  /// and `on_found`, which will (re-)initialize the bucket with the given key.
705  ///
706  /// PRECONDITION: hash_code == hash_(key)
707  template <class OnUninit, class OnInit, class OnFound>
708  void insert_helper_(size_t hash_code, const key_type& key,
709  OnUninit on_uninit, OnInit on_init, OnFound on_found)
710  {
711  maybe_grow_();
712 
713  size_t pos = which_bucket_(hash_code);
714  size_t dist = 0;
715 
716  for (;;) {
717  Bucket& bucket = buckets_[pos];
718 
719  if (!bucket.used_) {
720  on_uninit(bucket);
721  bucket.hash_code_ = hash_code;
722  ++size_;
723  return;
724  }
725 
726  auto bucket_locked = bucket.value_.lock();
727  auto bucket_key = weak_trait::key(bucket_locked);
728  if (!bucket_key) {
729  on_init(bucket);
730  bucket.hash_code_ = hash_code;
731  return;
732  }
733 
734  // If not expired, but matches the value to insert, replace.
735  if (hash_code == bucket.hash_code_ && equal_(key, *bucket_key)) {
736  on_found(bucket);
737  return;
738  }
739 
740  // Otherwise, we check the probe distance.
741  size_t existing_distance =
742  probe_distance_(pos, which_bucket_(bucket.hash_code_));
743  if (dist > existing_distance) {
744  steal_(bucket.hash_code_, next_bucket_(pos),
745  weak_trait::move(bucket_locked));
746  on_init(bucket);
